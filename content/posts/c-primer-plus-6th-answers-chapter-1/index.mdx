```cpp
#include <stdio.h>
#define INCH_PER_CM 2.54

int main(void) {
 
	double inch, cm;
	printf("Enter an inch value:");
	scanf("%lf", &inch);
	
	cm = inch * INCH_PER_CM;
	printf("The cm value is:%lf\n", cm);
	
	return 0;
}
```
