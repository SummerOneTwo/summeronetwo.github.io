4.2.c

```cpp
#include <stdio.h>
#include <string.h>

int main(void) {
	
	char first_name[20], last_name[20];
	printf("Please enter your first name.\n");
	scanf("%s", first_name);
	printf("Please enter your last name.\n");
	scanf("%s", last_name);
	
	printf("a: \"%s %s\"\n", first_name, last_name);
	printf("b: \"%20s %20s\"\n", first_name, last_name);
	printf("c: \"%-20s %-20s\"\n", first_name, last_name);
	
	int lenght_first, lenght_last;
	lenght_first = strlen(first_name);
	lenght_last = strlen(last_name);
	printf("d: %*s %*s", lenght_first, first_name, lenght_last, last_name);
	
	return 0;
}
```

4.6.c

```cpp
#include <stdio.h>

	char first_name[20], last_name[20];

	printf("Please enter your first name.\n");

	scanf("%s", last_name);
	
	lenght_first = strlen(first_name);
	lenght_last = strlen(last_name);

	printf("%-*d %-*d\n", lenght_first, lenght_first, lenght_last, lenght_last);
	
	return 0;
}
```
