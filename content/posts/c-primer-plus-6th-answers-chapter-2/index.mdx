2.3.c

```cpp
#include <stdio.h>
#define YEAR_PER_DAY 365

int main(void) {
	
	int age, days;
	age = 18;
	days = age * YEAR_PER_DAY;
	printf("age:%d\n", age);
	printf("days:%d\n", days);
	
	return 0;
}
```
