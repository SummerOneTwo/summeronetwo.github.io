6.5.c

```cpp
#include <stdio.h>
int main(void)
{
	char letter;
	
	printf("Please enter a capital letter.\n");
	scanf("%c", &letter);
	
	const int n = letter - 'A';
	
	int space, j, row;
	for(space = n, row = 0; space >= 0; space--, row++)
	{
		for(j = 0; j < space; j++)
		{
			printf(" ");
		}
		for(j = 0; j <= row; j++)
		{
			printf("%c", 'A' + j);
		}
		for(j = row; j > 0; j--)
		{
			printf("%c", 'A' + j - 1);
		}		
		printf("\n");
	}
	return 0;
}
```
